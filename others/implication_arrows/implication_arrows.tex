\section{בקצרה}

בבדיקת תרגילים בקורסים שונים בטכניון, ניתן לראות בעיה נפוצה מאוד בנוגע לחוסר הבנה כיצד להשתמש בחצים ובמיוחד בחצי גרירה.

התשובה הקצרה ביותר בנוגע לאופן השימוש בחצים היא שיש להשתמש בהם רק בהקשר של סימונים שנלמדו בקורס הרלוונטי, ושעדיף לכתוב הכל בלי חצי גרירה כלל אם לא בטוחות איך לעשות זאת. אפשר לכתוב כל טיעון והוכחה מתמטיים בלי שימוש בחצי גרירה, ובדרך כלל פתרונות של סטודנטים היו טובים יותר אם הם לא היו כוללים חצים כאלו.

\section{חצים רגילים}

בחצים רגילים, כמו
$\leftarrow, \rightarrow, \leftrightarrow$
נשתמש במסגרת הקורס רק בשני הקשרים.

\begin{enumerate}
\item
עבור קבוצות
$X,Y$
במקום לכתוב
\begin{quote}
תהי
$f$
פונקציה מ־%
$X$
ל־%
$Y$
כך שלכל
$x \in X$
מתקיים
\[\text{.} f\prs{x} = \text{\textenglish{some expression in $x$}}\]
\end{quote}
נכתוב
\begin{quote}
תהי
\begin{align*}
f \colon X &\to Y \\
\text{.} \hphantom{lala} x &\mapsto \text{\textenglish{some expression in $x$}}
\end{align*}
\end{quote}
\item
בדירוג מטריצות, נסמן פעולות דירוג בעזרת חץ
$\rightarrow$
עם תיאור מעליו של הפעולה המתאימה. למשל, ניתן לכתוב את זה:
\begin{quote}
\begin{align*}
\pmat{1 & 1 & 0 \\ 0 & 1 & 1 \\ 0 & 0 & 1} &\xrightarrow{R_2 \mapsto R_2 - R_3}
\pmat{1 & 1 & 0 \\ 0 & 1 & 0 \\ 0 & 0 & 1}
\\&\xrightarrow{R_1 \mapsto R_1 - R_2}
\pmat{1 & 0 & 0 \\ 0 & 1 & 0 \\ 0 & 0 & 1}
\end{align*}
לכן
$J_3\prs{1}$
שקולת שורה ל־%
$I_3$
ולכן
$\mathrm{rank}\prs{J_2\prs{1}} = 3$.
\end{quote}

יש לשים לב כי אין להשתמש בחצים כאלו או אחרים כדי לסמן שוויון, וכי המטריצות עם החצים בינן כאן רק מתארות את תהליך הדירוג.
\end{enumerate}

כל שימוש אחר בחצים כאלו אינו תקין, ויש במקומו לכתוב הסבר מפורש. למשל, במקום לכתוב
\begin{quote}
\[
\begin{tikzcd}[column sep=small]
&
\prs{D^2 - I}^n = (D+I)^n(D-I)^n=0
\arrow[dl]
\arrow[dr]
&
\\
\begin{array}{c}
(D+I)^n=0\\
\Downarrow\\
\text{$D+I$ הפיכה}\\
\text{ולכן זה לא יתכן}
\end{array}
&&
\begin{array}{c}
(D-I)^n=0\\
\Downarrow\\
n=4
\end{array}
\end{tikzcd}
\]
\end{quote}
מה שהופיע בערך במספר פתרונות בגיליון 6, יש לכתוב במפורש מה מתקיים, למשל כך:
\begin{quote}
נחפש את
$n \in \NN$
המינימלי עבורו
$\prs{D^2 - I}^n = 0$.
מתקיים
\begin{align*}
\text{,} \prs{D^2 - I}^n = \prs{\prs{D + I}\prs{D - I}}^n = \prs{D+I}^n \prs{D-I}^n
\end{align*}
וכיוון של־%
$D$
ערך עצמי יחיד
$1$,
האופרטור
$D+I$
הפיך, ולכן הביטוי
$\prs{D^2 - I}^n$
מתאפס אם ורק אם
$\prs{D - I}^n = 0$.

ראינו בסעיף קודם שהערך
$n \in \NN_+$
המינימלי המקיים זאת הוא
$4$,
ולכן זה גם הערך המינימלי עבורו
$\prs{D^2 - I}^n = 0$.
\end{quote}

שימו לב שגם השימוש בחצי הגרירה
$\Downarrow$
בניסוח הראשון אינו תקין. משמאל, הכתוב בסוף החץ אינו נובע מהכתוב למעלה, ומימין, חסר הסבר בנוגע למעבר, והוא גם אינו נכון במסגרת השאלה בה
$\prs{D-I}^4 = 0$
אך גם
$\prs{D-I}^n = 0$
לכל
$n > 4$.

\section{חצי גרירה}

בחצי גרירה כמו
$\implies, \impliedby, \iff$
עדיף לא להשתמש כלל, ואין צורך להשתמש בהם בכתיבת מתמטיקה. בכל זאת, נתאר כיצד כן אפשר להשתמש בהם עבור מי שיתעקשו לעשות זאת.

הביטוי
$X \implies Y$
משמעותו "$X$ גורר את $Y$", וניתן להשתמש בו אך ורק כך ואך ורק כאשר
$X,Y$
הינם פסוקים מתמטיים (כלומר, ביטויים מתמטיים הכוללים שוויונים, אי־שוויונים, שייכות, הכלה, וכו', שיכולים להיות נכונים או שאינם נכונים).

למשל, ניתן לכתוב:
\begin{quote}
משפט ריס למרחבי מכפלה פנימית סוף־מימדיים אומר לנו כי
\begin{align*}
\text{.} \phi \in V^* \implies \exists w \in V \, \forall v \in V: \phi\prs{v} = \trs{v, w}
\end{align*}
\end{quote}
או:
\begin{quote}
עבור הטלה
$P \in \End_\FF\prs{V}$
נסמן
$U = \im\prs{P}$
ו־%
$W = \ker\prs{P}$.
אז מתקיים
\begin{align*}
P = P^*
&\iff
\forall v_1, v_2 \in V: \trs{P\prs{v_1}, v_2} = \trs{v_1, P\prs{v_2}}
\\&\iff
\forall u_1, u_2 \in U, \, w_1, w_2 \in W: \trs{u_1, u_2 + w_2} = \trs{u_1 + w_1, u_2}
\\&\iff
\forall u_1, u_2 \in U, \, w_1, w_2 \in W: \trs{u_1, u_2} + \trs{u_1, w_2} = \trs{u_1, u_2} + \trs{w_1, u_2}
\\&\iff
\forall u_1, u_2 \in U, \, w_1, w_2 \in W: \trs{u_1, w_2} = \trs{w_1, u_2}
\\&\iff
\forall u \in U, w \in W: \trs{u,w} = 0
\\&\iff
W = U^\perp
\end{align*}
כאשר את הגרירה מטה בשורה השלישית מהסוף נקבל על ידי בחירת
$u_1 = u, w_2 = w, u_2 = w_1 = 0$.

נקבל בסך הכל כי
$P$
צמודה לעצמה אם ורק אם הגרעין ניצב לתמונה.
\end{quote}
או
\begin{quote}
ראינו כי עבור הטלה
$P$
על
$U \leq V$
מתקיים
\begin{align*}
\text{.} \text{$P$ צמודה לעצמה} \iff \text{$P$ נורמלית} \iff \text{$P = P_U$}
\end{align*}
\end{quote}

אבל, לא ניתן:
\begin{enumerate}
\item לכתוב גרירה שבאחד הקצוות שלה אין פסוק מתמטי.
למשל, גרירה לא יכולה להתחיל או להסתיים בקבוצה, מרחב וקטורי, מטריצה, אופרטור או פונקציה.

\item לכתוב גרירה שבה מה שנרצה להראות הוא לא הגרירה עצמה. למשל, במקום:
\begin{quote}
$A$
הפיכה ולכן
\begin{align*}
\adj\prs{A} A = \det\prs{A} \implies \adj\prs{A} = \det\prs{A} A^{-1}
\end{align*}
\end{quote}
כאשר צריך להראות כי
$\adj\prs{A} = \det\prs{A} A^{-1}$,
יש לכתוב זאת כך:
\begin{quote}
$A$
הפיכה וידוע מההרצאה כי
$\adj\prs{A} A = \det\prs{A}$.
נכפול את השוויון ב־%
$A^{-1}$
מימין ונקבל כי
\begin{align*}
\text{.} \adj\prs{A} = \det\prs{A} A^{-1}
\end{align*}
\end{quote}
מה שחשוב כאן הוא לא שמתקיימת הגרירה, אלא שמתקיימת המסקנה.

\item לכתוב חצי גרירה (או חצים אחרים) בלי הסברים ובמקום הסברים.
למשל, בדוגמה למעלה בה הראנו כי
$P = P^*$
אם ורק אם
$W = U^\perp$,
ההסבר בנוגע לאחת הגרירות נחוץ. באופן כללי, יש להסביר כל מעבר שאינו נובע מיד ובאופן פשוט מההגדרות, כאשר יש שימוש בחצים וכאשר אין.

\item לכתוב חץ גרירה כאשר הוא אינו מתאר את זה שמה שבתחילת החץ גורר את מה שבסופו. ראו את הדוגמה על
$\prs{D+I}^n$
למעלה.

\item להשתמש בחצי גרירה כאמצעי לכתוב מה נובע ממה ולא כאשר נרצה להגיד שמתקיימת הגרירה עצמה.

למשל, כאשר נתון
$X$
וצריך להוכיח
$Z$
במקום לכתוב:
\begin{quote}
\begin{align*}
X \implies Y \implies Z
\end{align*}
\end{quote}
יש לכתוב הסבר כמו:
\begin{quote}
נתון
$X$.
לפי ההגדרה%
\textbackslash%
משפט%
\textbackslash%
טענה מההרצאה מתקיים
$Y$,
וכפי שראינו בתרגול נובע מכך
$Z$.
\end{quote}
\end{enumerate}
